let currentSection = 0;
let isEnglish = true;

function navigateToSection(index) {
    const sections = document.querySelectorAll('.section');
    const navRight = document.getElementById('navRight');

    if (index === 0) {
        navRight.classList.remove('hidden');
    } else {
        navRight.classList.add('hidden');
    }

    sections[currentSection].classList.remove('active');
    sections[currentSection].classList.add('inactive');
    sections[index].classList.remove('inactive');
    sections[index].classList.add('active');
    currentSection = index;

    const titles = sections[index].querySelectorAll('.title, .subtitle, .text-block');
    titles.forEach(title => title.classList.add('visible'));

    // Обновите заголовок страницы в зависимости от текущей секции
    const sectionTitles = ['Home', 'About me', 'Portfolio', 'Contact']; // Добавьте "Contact"
    document.title = sectionTitles[index];
}

        function toggleTheme() {
    document.body.classList.toggle('light-theme');

    const darkVideo = document.getElementById('dark-theme-video');
    const lightVideo = document.getElementById('light-theme-video');

    if (document.body.classList.contains('light-theme')) {
        darkVideo.style.opacity = '0';
        setTimeout(() => darkVideo.style.display = 'none', 800); // Скрываем видео через 0.8 секунд
        lightVideo.style.display = 'block';
        setTimeout(() => lightVideo.style.opacity = '1', 0); // Плавное появление видео
    } else {
        lightVideo.style.opacity = '0';
        setTimeout(() => lightVideo.style.display = 'none', 800);
        darkVideo.style.display = 'block';
        setTimeout(() => darkVideo.style.opacity = '1', 0);
    }
}

        function toggleLanguage() {
            isEnglish = !isEnglish;
            translatePage(isEnglish);
        }

        function translatePage(isEnglish) {
    const elements = document.querySelectorAll('[data-en]');

    elements.forEach(el => {
        el.style.transition = 'opacity 0.5s';
        el.style.opacity = 0;

        setTimeout(() => {
            const text = isEnglish ? el.getAttribute('data-en') : el.getAttribute('data-ru');
            el.innerHTML = text; // Используйте innerHTML вместо textContent

            el.style.opacity = 1;
        }, 500);
    });

    document.getElementById('language-icon').src = isEnglish ? '3.png' : '4.png';
}


window.addEventListener('wheel', (event) => {
    if (event.deltaY > 0 && currentSection < 3) {
        navigateToSection(currentSection + 1);
    } else if (event.deltaY < 0 && currentSection > 0) {
        navigateToSection(currentSection - 1);
    }
});

document.addEventListener('DOMContentLoaded', () => {
            // Функция для добавления обработчика клика на элемент
            function addClickHandler(className) {
                const element = document.querySelector(`.${className}`);
                
                if (element) {
                    element.addEventListener('click', () => {
                        if (className === 'rounded-border11') {
                            // Обработка изменения цвета для элемента с классом 'rounded-border11'
                            element.classList.toggle('active');
                        } else {
                            // Переход по URL для остальных элементов
                            const url = element.getAttribute('data-url');
                            if (url) {
                                window.location.href = url;
                            }
                        }
                    });
                }
            }

            // Добавляем обработчики для всех элементов
            addClickHandler('rounded-border4');
            addClickHandler('rounded-border5');
            addClickHandler('rounded-border6');
            addClickHandler('rounded-border11');
        });
